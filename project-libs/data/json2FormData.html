    # json2FormData 方法

    接口 function json2FormData(obj: any): FormData;

    版本 1.0.0

    # 作用：对象转为 formdata

    // 引入
import { json2FormData } from 'project-libs';

// 单独引入
import json2FormData from 'project-libs/build/data/json2FormData';

    # 参数

    参数 类型 是否必须 含义
    obj object 是 对象

    # 返回值

    是否有返回值 类型 说明
    是 FormData 返回FormData

    # 小例子

    // 使用
var params = {
  id: 10,
  name: 'admin'
};

var fd = json2FormData(params);
for (var [a, b] of fd.entries()) {
  console.log(a, b);
}

// id 10
// name admin
