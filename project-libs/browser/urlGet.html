    # urlGet 方法

    接口 function urlGet(key: string, href?: string): string | null;

    版本 1.0.0

    # 作用：获得URL中GET请求的参数值

    # 引入

    // 直接引入
import { urlGet } from 'project-libs';

// 单独引入
import urlGet from 'project-libs/build/browser/urlGet';

    # 参数

    参数 类型 是否必须 含义
    key string 是 参数
    href string 否 网址，如果没有则默认选取当前网址

    # 返回值

    是否有返回值 类型 说明
    是 string 如果有值返回字符串，否则返回 null

    # 小例子

    // href 为 http://localhost:8080/?id=10
urlGet('id')
// 10
