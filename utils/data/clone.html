    # clone 方法

    接口 function clone(source: any): any;

    版本 1.0.0

    # 作用：浅拷贝，才方法只针对普通对象{}和数组[]

    // 引入
import { clone } from '@cpage/utils';

// 单独引入
import clone from '@cpage/utils/build/data/clone';

    # 参数

    参数 类型 是否必须 含义
    source array 或 object 是 需要拷贝的元素

    # 返回值

    是否有返回值 类型 说明
    是 array 或 object 返回一个新的数组或对象

    # 小例子

    // 使用
const obj = {id: 10};
const newObj = clone(obj);

// {id: 10}
